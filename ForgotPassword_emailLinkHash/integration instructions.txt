SETUP:
You will need to change the DB_PASSWORD and the MAIL_LINK = "http://localhost:8080/NewCapeWatch/VerifyEmailHash";


web.xml:
	- servlet declaration & mapping needs to be added to NewCapeWatch web.xml from this web.xml

	<servlet>
			<servlet-name>Forgot Password</servlet-name>
			<servlet-class>forgotPassword.servlets.ForgotPassword</servlet-class>
	</servlet>
	<servlet>
			<servlet-name>login</servlet-name>
			<servlet-class>forgotPassword.servlets.login</servlet-class>
	</servlet>
	<servlet-mapping>
			<servlet-name>Forgot Password</servlet-name>
			<url-pattern>/ForgotPassword</url-pattern>
	</servlet-mapping>
	<servlet>
			<servlet-name>Change Password</servlet-name>
			<servlet-class>forgotPassword.servlets.ChangePassword</servlet-class>
	</servlet>
	<servlet-mapping>
			<servlet-name>Change Password</servlet-name>
			<url-pattern>/createPass</url-pattern>
	</servlet-mapping>
	<servlet>
			<servlet-name>Verify Email Hash</servlet-name>
			<servlet-class>forgotPassword.servlets.VerifyEmailHash</servlet-class>
	</servlet>
	<servlet-mapping>
			<servlet-name>Verify Email Hash</servlet-name>
			<url-pattern>/VerifyEmailHash</url-pattern>
	</servlet-mapping>
	<servlet>
			<servlet-name>Register User</servlet-name>
			<servlet-class>forgotPassword.servlets.RegisterUser</servlet-class>
	</servlet>
	<servlet-mapping>
			<servlet-name>Register User</servlet-name>
			<url-pattern>/register</url-pattern>
	</servlet-mapping>
	<servlet>
			<servlet-name>Login</servlet-name>
			<servlet-class>forgotPassword.servlets.login</servlet-class>
	</servlet>
	<servlet-mapping>
			<servlet-name>Login</servlet-name>
			<url-pattern>/login</url-pattern>
	</servlet-mapping>
